900 !**********************************************************************!
901 subroutine assvmax2(a,w,pc,gs,po,pa,vmax,g0,g1,rh,ko,kc,kg,tau,rd,&
902  acheck)
903 !**********************************************************************!
904 real(dp) :: a,w,pc,gs,po,pa,vmax,g0,g1,rh,ko,kc,kg,tau,rd,acheck,r2q3, &
905  ax,bx,cx,dx,ex,fx,gx,hx,p,q,r,qt,rt,th,a1,a2,a3,pi,at,bt,sc,arg1
906 !----------------------------------------------------------------------!
907 
908 pi = atan(1.0)*4.0
909 sc = 1000000.0
910 !----------------------------------------------------------------------!
911 ! 'a2,b2,c2,d2' are such that 'w=(a2a + b2)/(c2a + d2)'. !
912 !----------------------------------------------------------------------!
913 ax = ko*pa*vmax*(kg*g1*rh - 160.0)
914 bx = kg*g0*pa**2.0*vmax*ko
915 cx =-160.0*pa*ko + kc*kg*ko*g1*rh + pa*kg*ko*g1*rh + kc*kg*po*g1*rh
916 dx = pa**2.0*kg*ko*g0 + kc*kg*po*g0*pa + kc*kg*ko*g0*pa
917 
918 !----------------------------------------------------------------------!
919 ! 'ex,fx,gx,hx' are such that '0.5po/(taupc)-1=(exa+fx)/(gxa+hx)'. !
920 !----------------------------------------------------------------------!
921 ex =-tau*pa*kg*g1*rh + 160.0*tau*pa + 0.5*po*kg*g1*rh
922 fx =-tau*pa**2*kg*g0 + 0.5*po*kg*g0*pa
923 gx = tau*pa*(kg*g1*rh - 160.0)
924 hx = tau*pa**2.0*kg*g0
925 
926 !----------------------------------------------------------------------!
927 ! Cubic coefficients 'p,q,r'. !
928 !----------------------------------------------------------------------!
929 p = sc*rd + (sc*ax*ex + cx*hx + dx*gx)/(cx*gx)
930 q = (sc*rd*cx*hx + sc*rd*dx*gx + sc*ax*fx + sc*bx*ex + dx*hx)/(cx*gx)
931 r = (sc*rd*dx*hx + sc*bx*fx)/(cx*gx)
932 
933 !----------------------------------------------------------------------!
934 ! Sove the cubic equaiton to find 'a'. !
935 !----------------------------------------------------------------------!
936 qt = (p**2 - 3.0*q)/9.0
937 rt = (2.0*p**3 - 9.0*p*q + 27.0*r)/54.0
938 
939 r2q3 = rt**2 - qt**3
940 
941 if (r2q3<0.0) then
942  arg1 = rt/qt**1.5
943  if (arg1>1.0) arg1 = 1.0
944  if (arg1<-1.0) arg1 = -1.0
945  th = acos(arg1)
946  a1 =-2.0*qt**0.5*cos(th/3.0) - p/3.0
947  a2 =-2.0*qt**0.5*cos((th + 2.0*pi)/3.0) - p/3.0
948  a3 =-2.0*qt**0.5*cos((th + 4.0*pi)/3.0) - p/3.0
949  a = a1
950  if (a2>a) a = a2
951  if (a3>a) a = a3
952 else
953  at =-rt/abs(rt)*(abs(rt) + r2q3**0.5)**(1.0/3.0)
954  if (abs(at)>1e-6) then
955  bt = qt/at
956  else
957  bt = 0.0
958  endif
959  a1 = at + bt - p/3.0
960 ! a2 =-0.5*(at + bt) - p/3.0 + i*3.0**0.5*(at - bt)/2.0
961 ! a3 =-0.5*(at + bt) - p/3.0 - i*3.0**0.5*(at - bt)/2.0
962  a = a1
963 endif
964 
965 !----------------------------------------------------------------------!
966 ! Compute 'gs,pc,w' corresponding to 'a'. !
967 !----------------------------------------------------------------------!
968 gs = (g0 + g1*a*rh/pa)*kg
969 pc = pa - a*160.0/gs
970 w = vmax*pc/(pc + kc*(1 + po/ko))
971 acheck = (w*(1.0 - 0.5*po/(tau*pc)) - rd)*sc
972 
973 !**********************************************************************!
974 end subroutine assvmax2
